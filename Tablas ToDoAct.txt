-- ============================================
-- TABLAS ORIGINALES
-- ============================================

CREATE TABLE usuarios(
	id SERIAL PRIMARY KEY,
	usuario VARCHAR(50) UNIQUE NOT NULL,
	contraseña VARCHAR(50) UNIQUE NOT NULL,
	nombrecompleto VARCHAR(255) NOT NULL,
	fechanacimiento DATE NOT NULL,
	genero VARCHAR(20) NOT NULL
);

CREATE TABLE actividades(
	id INT PRIMARY KEY,
	actividad VARCHAR(255) UNIQUE NOT NULL
);

CREATE TABLE niveles(
	id SERIAL NOT NULL PRIMARY KEY,
	id_usuario INT NOT NULL,
	id_actividad INT NOT NULL,
	categoria TEXT DEFAULT NULL,
	nivel INT NOT NULL,
	desbloqueado BOOLEAN DEFAULT false,
	FOREIGN KEY (id_usuario) REFERENCES usuarios(id),
	FOREIGN KEY (id_actividad) REFERENCES actividades(id)
);

CREATE INDEX niveles_idx ON niveles (id_usuario, id_actividad, categoria, nivel);

CREATE TABLE puntuaciones(
	id SERIAL PRIMARY KEY,
	id_usuario INT NOT NULL,
	id_actividad INT NOT NULL,
	categoria TEXT,
	nivel INT NOT NULL,
	puntuacion INT NOT NULL,
	tiempo INT NOT NULL,
	errores INT NOT NULL,
	fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	FOREIGN KEY (id_usuario) REFERENCES usuarios(id),
	FOREIGN KEY (id_actividad) REFERENCES actividades(id)
);

-- ============================================
-- MODIFICACIONES A TABLAS EXISTENTES
-- ============================================

ALTER TABLE usuarios ADD COLUMN estilo_aprendizaje VARCHAR(30);
ALTER TABLE puntuaciones ADD COLUMN estilo_mostrado VARCHAR(30);

-- ============================================
-- TABLAS DEL SISTEMA DE CONTENIDOS
-- ============================================

CREATE TABLE contenidos (
	id SERIAL PRIMARY KEY,
	area VARCHAR(50) NOT NULL,
	slug VARCHAR(100) NOT NULL,
	nombre VARCHAR(150) NOT NULL,
	orden INT DEFAULT 0,
	activo BOOLEAN DEFAULT true
);

CREATE TABLE niveles_contenido (
	id SERIAL PRIMARY KEY,
	contenido_id INTEGER NOT NULL REFERENCES contenidos(id) ON DELETE CASCADE,
	nivel VARCHAR(20) NOT NULL
);

CREATE TABLE progreso_contenido_usuario (
	usuario VARCHAR(50) NOT NULL,
	contenido_id INTEGER NOT NULL REFERENCES contenidos(id) ON DELETE CASCADE,
	diagnostico_realizado BOOLEAN DEFAULT false,
	nivel_asignado VARCHAR(20),
	contenido_completado BOOLEAN DEFAULT false,
	fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	tiempo_total_diagnostico INTEGER,
	fecha_inicio_diagnostico TIMESTAMP,
	fecha_fin_diagnostico TIMESTAMP,
	PRIMARY KEY (usuario, contenido_id)
);

CREATE TABLE contenidos_archivos (
	contenido_id INTEGER NOT NULL REFERENCES contenidos(id) ON DELETE CASCADE,
	nivel VARCHAR(20) NOT NULL,
	estilo VARCHAR(30) NOT NULL,
	archivo TEXT NOT NULL,
	PRIMARY KEY (contenido_id, nivel, estilo)
);

CREATE TABLE examenes_diagnostico (
	id SERIAL PRIMARY KEY,
	contenido_id INTEGER NOT NULL REFERENCES contenidos(id) ON DELETE CASCADE,
	activo BOOLEAN DEFAULT true
);

CREATE TABLE preguntas_diagnostico (
	id SERIAL PRIMARY KEY,
	examen_id INTEGER NOT NULL REFERENCES examenes_diagnostico(id) ON DELETE CASCADE,
	pregunta TEXT NOT NULL,
	opcion_a TEXT NOT NULL,
	opcion_b TEXT NOT NULL,
	opcion_c TEXT NOT NULL,
	opcion_d TEXT NOT NULL,
	respuesta_correcta CHAR(1) NOT NULL
);

CREATE TABLE respuestas_diagnostico (
	id SERIAL PRIMARY KEY,
	usuario VARCHAR(50) NOT NULL,
	contenido_id INTEGER NOT NULL REFERENCES contenidos(id) ON DELETE CASCADE,
	pregunta_id INTEGER NOT NULL REFERENCES preguntas_diagnostico(id) ON DELETE CASCADE,
	respuesta_dada CHAR(1) NOT NULL,
	es_correcta BOOLEAN NOT NULL,
	intento INTEGER NOT NULL DEFAULT 1,
	fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_resp_diag_usuario ON respuestas_diagnostico(usuario);
CREATE INDEX idx_resp_diag_contenido ON respuestas_diagnostico(contenido_id);

CREATE TABLE resultados_quiz (
	id SERIAL PRIMARY KEY,
	usuario VARCHAR(50) NOT NULL,
	contenido_id INTEGER NOT NULL REFERENCES contenidos(id) ON DELETE CASCADE,
	nivel VARCHAR(20) NOT NULL,
	estilo VARCHAR(30) NOT NULL,
	aciertos INTEGER NOT NULL DEFAULT 0,
	total_preguntas INTEGER NOT NULL DEFAULT 5,
	puntaje INTEGER NOT NULL DEFAULT 0,
	tiempo_total INTEGER,
	intento INTEGER NOT NULL DEFAULT 1,
	aprobado BOOLEAN NOT NULL DEFAULT false,
	fecha_inicio TIMESTAMP,
	fecha_fin TIMESTAMP,
	fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_quiz_usuario ON resultados_quiz(usuario);
CREATE INDEX idx_quiz_contenido ON resultados_quiz(contenido_id);
CREATE INDEX idx_quiz_nivel ON resultados_quiz(nivel);
CREATE INDEX idx_quiz_aprobado ON resultados_quiz(aprobado);
